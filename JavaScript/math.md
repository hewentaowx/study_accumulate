## 有关math的相关语法

```javascript
/** FIXME: 常用的 */
// Math.cos函数  返回数字的余弦值
console.log('Math.cos函数返回数字的余弦值--->>>', Math.cos(-1))

// Math.exp函数  返回 e (自然对数的底)的乘幂数
console.log('Math.exp函数返回 e 的乘幂数--->>>', Math.exp(-1))

// Math.floor函数  返回小于或等于提供的数值表达式的最大整数
console.log('Math.floor函数返回小于或等于提供数值表达式的最大整数--->>>', Math.floor(-1))

// Math.hypot(value[, vakue2[, value3]])函数  返回参数平方和的平方根 value1必需
console.log('Math.hypot函数返回参数平方和的平方根--->>>', Math.hypot(-1, 3))

// Math.imul函数  返回被视为 32 位带符号整数的两个数字的积
console.log('Math.imul函数返回被视为 32 位带符号整数的两个数字的积--->>>', Math.imul(-1, 9));

// Math.log函数  返回数字的自然对数
console.log('Math.log函数返回数字的自然对数--->>>', Math.log(8) / Math.log(2))

// Math.log10函数  返回数字以10为底的对数
console.log('Math.log10函数返回数字以10为底的对数--->>>', Math.log10(10000))

// Math.log2函数  返回数字以2为底的对数
console.log('Math.log2函数返回数字以2为底的对数--->>>', Math.log2(8))

// Math.max函数  返回提供的两个数值表达式中的较大值。
console.log('Math.max返回提供的两个数值表达式中的较大值。--->>>', Math.max(5, 9))

// Math.min函数  返回提供的两个数值表达式中的较小值。
console.log('Math.min返回提供的两个数值表达式中的较小值。--->>>', Math.min(5, 9))

// Math.pow函数  返回基表达式的指定乘幂数的值
console.log('Math.pow返回基表达式的指定乘幂数的值。--->>>', Math.pow(4, 0.5))

// Math.random函数  返回介于 0 和 1 之间的伪随机数。
console.log('Math.random返回介于 0 和 1 之间的伪随机数。--->>>', Math.random())

// Math.round函数  返回舍入到最近整数的指定数值表达式。
console.log('Math.round返回舍入到最近整数的指定数值表达式。--->>>', Math.round(3.1))

// Math.sign函数  返回数字符号，它指示数字为正数、负数还是 0
console.log('Math.sign返回数字符号，它指示数字为正数、负数还是 0。--->>>', Math.sign(3.1))

// Math.sin函数  返回数字的正弦值
console.log('Math.sin返回数字的正弦值--->>>', Math.sin(3.1))

// Math.tan函数  返回数字的正切值
console.log('Math.tan返回数字的正切值--->>>', Math.tan(3.1))

// Math.sqrt函数  返回数字的平方根
console.log('Math.sqrt返回数字的平方根--->>>', Math.sqrt(4))

// Math.trunc函数  返回数字的整数部分，删除任何小数数字
console.log('Math.trunc返回数字的整数部分，删除任何小数数字--->>>', Math.trunc(4.1))

/** FIXME: 不常用的 */
// Math.abs函数  返回数字的绝对值
console.log('Math.abs函数返回数字的绝对值--->>>', Math.abs(-1))

// Math.cos函数 返回一个数字的余弦值
console.log('Math.cos函数返回数字的余弦值--->>>', Math.cos(-1.0))

// Math.acos函数 返回一个数字的反余弦(逆余弦)
console.log('Math.acos函数返回数字的反余弦值--->>>', Math.acos(-1.0))

// Math.acosh函数 返回一个数字的双曲反余弦值(或反双曲余弦值)
console.log('Math,acosh函数返回数字的双曲反余弦值或反双曲余弦值--->>>', Math.acosh(3))

// Math.asin函数 返回一个数字的反正弦值
console.log('Math.asin函数返回一个数字的反正弦值--->>>', Math.asin(1))

// Math.asinh函数 返回一个数字的反双曲正弦值
console.log('Math.asinh函数返回一个数字的反双曲正弦--->>>', Math.asinh(1))

// Math.atan函数 返回一个数字的反正切值
console.log('Math.atan函数返回数字的反正切值--->>>', Math.atan(1))

// Math.atan2函数 将与X轴的角度以弧度为单位 返回从 x 轴到点 (x,y) 之间的角度 y x
console.log('Math.atan2函数表示坐标点--->>>', Math.atan2(0.5, 0.5))

// Math.atanh函数 返回一个数字的反双曲正切值
console.log('Math.atanh函数返回一个数字的反双曲正切值--->>>', Math.atanh(1))

// Math.cosh函数  返回数字的双曲余弦值
console.log('Math.cosh函数返回数字的双曲余弦值--->>>', Math.cosh(-1))

// Math.expm1函数  返回 e (自然对数的底)的乘幂数减去 1 的结果
console.log('Math.expm1函数返回 e 的乘幂数减去 1 的值--->>>', Math.expm1(-1))

// Math.log1p函数  返回 1 加上一个数字的自然对数
console.log('Math.log1p函数返回 1 加上一个数字的自然对数--->>>', Math.log1p(8))

// Math.sinh函数  返回数字的反双曲正弦
console.log('Math.sinh返回数字的反双曲正弦--->>>', Math.sinh(3.1))

// Math.tanh函数  返回数字的双曲正切
console.log('Math.tanh返回数字的双曲正切--->>>', Math.tanh(3.1))

```
